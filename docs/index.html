Functions

  • Parameters are not checked

    function fun1(arg) {
    return arg;
}

fun1(); // undefined
fun1(42, 35, "hello"); // 42

  • No function overloading

    function fun1(arg) {
    return arg;
}

fun1(); // undefined
fun1(42, 35, "hello"); // 42

  • return keyword can’t be omitted

    function noReturn() {
    1;
}

noReturn(); // undefined

  • Curly brackets constitute a function body

  • Functions can be freely bound to new variables

    const myfun = fun1;

  • Higher-order functions

    function fun1(fun2) {
    return function fun3(arg) { 
        return fun2(arg);
    };
}

  • Curried functions

    const plus = (x) => (y) => (x+y);
const intermediate = plus(3);

Lambda Expressions

const sum = function (a, b) { return a + b };

// can be written as:
const sum = (a, b) => a + b;

// or
const sum = (a, b) => { return a + b };

// parentheses optional for one parameters
const sum = a => a;

// without parameters
var noop = () => {} // noop

// application:
[1,2,3,4].filter(function (value) {return value % 2 === 0});
// to:
[1,2,3,4].filter(value => value % 2 === 0);
